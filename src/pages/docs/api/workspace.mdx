# Workspace API

The Workspace API provides discovery of project directories on the filesystem, bulk import into ainative with automatic environment scanning, and runtime context about the current workspace. The import endpoint uses SSE to stream progress as each project is created and scanned.

## Quick Start

Discover projects in a parent directory, then import the new ones with streamed progress:

<CodeExample ts={`// 1. Discover project directories under ~/Developer
interface DiscoveredProject {
  path: string;
  name: string;
  markers: string[];
  alreadyImported: boolean;
}

const { projects }: { projects: DiscoveredProject[] } = await fetch('/api/workspace/discover', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    parentDir: '~/Developer',
    maxDepth: 2,
    markers: ['package.json', 'Cargo.toml', 'go.mod'],
  }),
}).then((r: Response) => r.json());

// 2. Filter to projects not yet imported
const newProjects: DiscoveredProject[] = projects.filter((p) => !p.alreadyImported);
console.log(\`Found \${projects.length} projects, \${newProjects.length} new\`);

// 3. Import new projects with streamed progress (SSE)
const res: Response = await fetch('/api/workspace/import', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    projects: newProjects.map((p) => ({ path: p.path, name: p.name })),
  }),
});

const reader: ReadableStreamDefaultReader<Uint8Array> = res.body!.getReader();
const decoder: TextDecoder = new TextDecoder();

while (true) {
  const { done, value } = await reader.read();
  if (done) break;

  const text: string = decoder.decode(value);
  for (const line of text.split('\\n')) {
    if (line.startsWith('data: ')) {
      const event: { type: string; name?: string; status?: string; created?: number; failed?: number } =
        JSON.parse(line.slice(6));
      if (event.type === 'progress') {
        console.log(\`\${event.name}: \${event.status}\`);
      } else if (event.type === 'complete') {
        console.log(\`Done! \${event.created} imported, \${event.failed} failed\`);
      }
    }
  }
}`} python={`import requests
import json

# 1. Discover project directories under ~/Developer
discover_res = requests.post(
    "http://localhost:3000/api/workspace/discover",
    json={
        "parentDir": "~/Developer",
        "maxDepth": 2,
        "markers": ["package.json", "Cargo.toml", "go.mod"],
    },
)
projects: list[dict] = discover_res.json()["projects"]

# 2. Filter to projects not yet imported
new_projects: list[dict] = [p for p in projects if not p["alreadyImported"]]
print(f"Found {len(projects)} projects, {len(new_projects)} new")

# 3. Import new projects with streamed progress (SSE)
import_res = requests.post(
    "http://localhost:3000/api/workspace/import",
    json={
        "projects": [{"path": p["path"], "name": p["name"]} for p in new_projects],
    },
    stream=True,
)

for line in import_res.iter_lines(decode_unicode=True):
    if line.startswith("data: "):
        event: dict = json.loads(line[6:])
        if event["type"] == "progress":
            print(f"{event['name']}: {event['status']}")
        elif event["type"] == "complete":
            print(f"Done! {event['created']} imported, {event['failed']} failed")`} />

### Discover Projects

<EndpointCard method="POST" path="/api/workspace/discover" description="Scan a parent directory for project directories using configurable marker files (package.json, Cargo.toml, etc.). Flags projects that are already imported into ainative.">

<SchemaTable title="Request Body" fields={[
  { name: 'parentDir', type: 'string', required: true, description: 'Parent directory to scan (supports ~ expansion)' },
  { name: 'maxDepth', type: 'number', required: false, description: 'Maximum directory depth to search' },
  { name: 'markers', type: 'string[]', required: false, description: 'Marker filenames to detect projects (e.g., package.json, .git)' },
]} />

<SchemaTable title="Response Body" fields={[
  { name: 'projects', type: 'object[]', required: true, description: 'Discovered project directories' },
  { name: 'projects[].path', type: 'string', required: true, description: 'Absolute path to the project' },
  { name: 'projects[].name', type: 'string', required: true, description: 'Directory name' },
  { name: 'projects[].markers', type: 'string[]', required: true, description: 'Which marker files were found' },
  { name: 'projects[].alreadyImported', type: 'boolean', required: true, description: 'Whether this project already exists in ainative' },
]} />

**Errors:** `400` — Validation failure or directory not found

Scan a developer directory to find all project directories — already-imported projects are flagged so you can skip them:

<CodeExample ts={`// Discover project directories
interface DiscoveredProject {
  path: string;
  name: string;
  markers: string[];
  alreadyImported: boolean;
}

const { projects }: { projects: DiscoveredProject[] } = await fetch('/api/workspace/discover', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    parentDir: '~/Developer',
    maxDepth: 2,
  }),
}).then((r: Response) => r.json());

// Show discovered projects, highlighting new ones
projects.forEach((p: DiscoveredProject) => {
  const status: string = p.alreadyImported ? 'imported' : 'NEW';
  console.log(\`[\${status}] \${p.name} — \${p.path} (\${p.markers.join(', ')})\`);
});

const newProjects: DiscoveredProject[] = projects.filter((p) => !p.alreadyImported);
console.log(\`\${newProjects.length} new projects ready to import\`);`} python={`import requests

# Discover project directories
response = requests.post(
    "http://localhost:3000/api/workspace/discover",
    json={
        "parentDir": "~/Developer",
        "maxDepth": 2,
    },
)
data: dict = response.json()

# Show discovered projects, highlighting new ones
for p in data["projects"]:
    status: str = "imported" if p["alreadyImported"] else "NEW"
    markers: str = ", ".join(p["markers"])
    print(f"[{status}] {p['name']} — {p['path']} ({markers})")

new_projects: list[dict] = [p for p in data["projects"] if not p["alreadyImported"]]
print(f"{len(new_projects)} new projects ready to import")`} />

### Fix Data Directory

<EndpointCard method="POST" path="/api/workspace/fix-data-dir" description="Fix a data-directory mismatch for domain-cloned ainative instances. Derives the correct AINATIVE_DATA_DIR from the folder name, writes it to .env.local alongside AINATIVE_CLOUD_DISABLED=true, and bootstraps a fresh SQLite database at that path. A server restart is required to take effect.">

<SchemaTable title="Response Body" fields={[
  { name: 'success', type: 'boolean', required: true, description: 'Whether the fix was applied' },
  { name: 'dataDir', type: 'string', required: true, description: 'The derived data directory path (e.g. ~/.ainative-wealth)' },
  { name: 'envLocalPath', type: 'string', required: true, description: 'Absolute path to the .env.local file that was updated' },
  { name: 'needsRestart', type: 'boolean', required: true, description: 'Always true — the server must be restarted to load the new database path' },
]} />

**Errors:**
- `400` — The main dev repo does not need a fix, or `AINATIVE_DATA_DIR` is already set to a non-default path

Use this endpoint once after cloning ainative to a new folder name (e.g. `ainative-wealth`) to isolate its database from the main instance:

<CodeExample ts={`// Apply data-dir isolation for a domain clone — run once after cloning
const res: Response = await fetch('http://localhost:3000/api/workspace/fix-data-dir', {
  method: 'POST',
});

if (res.ok) {
  const result: { success: boolean; dataDir: string; envLocalPath: string; needsRestart: boolean } =
    await res.json();
  console.log(\`Data dir set to: \${result.dataDir}\`);
  console.log(\`Updated: \${result.envLocalPath}\`);
  if (result.needsRestart) {
    console.log('Restart the dev server to activate the new database path');
  }
} else {
  const { error }: { error: string } = await res.json();
  console.error(\`Fix not needed: \${error}\`);
}`} python={`import requests

# Apply data-dir isolation for a domain clone — run once after cloning
res: requests.Response = requests.post("http://localhost:3000/api/workspace/fix-data-dir")

if res.ok:
    result: dict = res.json()
    print(f"Data dir set to: {result['dataDir']}")
    print(f"Updated: {result['envLocalPath']}")
    if result["needsRestart"]:
        print("Restart the dev server to activate the new database path")
else:
    error_data: dict = res.json()
    print(f"Fix not needed: {error_data['error']}")`} />

### Import Projects (SSE)

<EndpointCard method="SSE" path="/api/workspace/import" description="Bulk-import discovered projects into ainative. Each project is created with an automatic environment scan. Progress is streamed via Server-Sent Events so the UI can update in real time.">

<SchemaTable title="Request Body (POST)" fields={[
  { name: 'projects', type: 'object[]', required: true, description: 'Array of projects to import' },
  { name: 'projects[].path', type: 'string', required: true, description: 'Absolute path to the project directory' },
  { name: 'projects[].name', type: 'string', required: true, description: 'Project name' },
]} />

**SSE Event Types:**

| Event type | Fields | Description |
|------------|--------|-------------|
| `progress` | `index`, `name`, `status: "creating"` | Import started for a project |
| `progress` | `index`, `name`, `status: "done"`, `projectId`, `artifactCount` | Project created and scanned |
| `progress` | `index`, `name`, `status: "failed"`, `error` | Import failed for a project |
| `complete` | `created`, `failed` | All imports finished |

**Errors:** `400` — Validation failure

Import multiple projects at once and stream progress to the UI — each project gets an environment scan automatically:

<CodeExample ts={`// Import projects with streamed progress
const res: Response = await fetch('/api/workspace/import', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    projects: [
      { path: '/Users/team/Developer/my-app', name: 'My App' },
      { path: '/Users/team/Developer/rust-cli', name: 'Rust CLI' },
    ],
  }),
});

// Read the SSE stream
const reader: ReadableStreamDefaultReader<Uint8Array> = res.body!.getReader();
const decoder: TextDecoder = new TextDecoder();

while (true) {
  const { done, value } = await reader.read();
  if (done) break;

  const text: string = decoder.decode(value);
  for (const line of text.split('\\n')) {
    if (line.startsWith('data: ')) {
      const event: Record<string, unknown> = JSON.parse(line.slice(6));

      if (event.type === 'progress') {
        if (event.status === 'creating') {
          console.log(\`Importing \${event.name}...\`);
        } else if (event.status === 'done') {
          console.log(\`  Done: \${event.projectId} (\${event.artifactCount} artifacts)\`);
        } else if (event.status === 'failed') {
          console.error(\`  Failed: \${event.error}\`);
        }
      } else if (event.type === 'complete') {
        console.log(\`Import complete: \${event.created} created, \${event.failed} failed\`);
      }
    }
  }
}`} python={`import requests
import json

# Import projects with streamed progress
response = requests.post(
    "http://localhost:3000/api/workspace/import",
    json={
        "projects": [
            {"path": "/Users/team/Developer/my-app", "name": "My App"},
            {"path": "/Users/team/Developer/rust-cli", "name": "Rust CLI"},
        ],
    },
    stream=True,
)

# Read the SSE stream
for line in response.iter_lines(decode_unicode=True):
    if line.startswith("data: "):
        event: dict = json.loads(line[6:])

        if event["type"] == "progress":
            if event["status"] == "creating":
                print(f"Importing {event['name']}...")
            elif event["status"] == "done":
                print(f"  Done: {event['projectId']} ({event['artifactCount']} artifacts)")
            elif event["status"] == "failed":
                print(f"  Failed: {event['error']}")
        elif event["type"] == "complete":
            print(f"Import complete: {event['created']} created, {event['failed']} failed")`} />
