# Instance API

The Instance API manages the ainative process lifecycle on a self-hosted installation. It exposes instance configuration (branch name, data directory, guardrails), idempotent initialization, and the full upgrade detection + application flow. The upgrade system uses a background poller that compares the local branch against `origin/main`; when commits are available, the operator spawns an `upgrade-assistant` task that performs a guided git merge.

When running on the canonical ainative development repository (`AINATIVE_DEV_MODE=true` or the `.git/ainative-dev-mode` sentinel), all endpoints return safe synthetic responses so that dev-environment state never bleeds into production UIs.

## Quick Start

Check for available upgrades, trigger the upgrade if one is found, and poll until the task completes:

<CodeExample ts={`// 1. Read current instance config and upgrade state
interface InstanceConfig {
  branchName: string;
  dataDir: string;
  createdAt: string;
}
interface UpgradeState {
  upgradeAvailable: boolean;
  commitsBehind: number;
  lastPolledAt: number | null;
  lastUpgradeTaskId: string | null;
}
interface ConfigResponse {
  devMode: boolean;
  config: InstanceConfig | null;
  guardrails: object | null;
  upgrade: UpgradeState | null;
}

const { devMode, config, upgrade }: ConfigResponse =
  await fetch('/api/instance/config').then((r: Response) => r.json());

if (devMode) {
  console.log('Running in dev mode — upgrade flow disabled');
} else if (!config) {
  // 2. Initialize if not yet bootstrapped
  await fetch('/api/instance/init', { method: 'POST' });
  console.log('Instance initialized');
} else {
  console.log(\`Branch: \${config.branchName}\`);
  console.log(\`Upgrade available: \${upgrade?.upgradeAvailable}\`);
  console.log(\`Commits behind: \${upgrade?.commitsBehind}\`);
}

// 3. Force-check for new upstream commits
const { ok, state }: { ok: boolean; state?: UpgradeState } =
  await fetch('/api/instance/upgrade/check', { method: 'POST' }).then((r: Response) => r.json());

if (ok && state?.upgradeAvailable) {
  // 4. Spawn the upgrade task
  const { taskId }: { taskId: string } =
    await fetch('/api/instance/upgrade', { method: 'POST' }).then((r: Response) => r.json());

  console.log(\`Upgrade task spawned: \${taskId}\`);
  console.log('Navigate to the task view to watch streaming progress');
}`} python={`import requests

# 1. Read current instance config and upgrade state
config_res = requests.get("http://localhost:3000/api/instance/config")
data: dict = config_res.json()

if data["devMode"]:
    print("Running in dev mode — upgrade flow disabled")
elif not data["config"]:
    # 2. Initialize if not yet bootstrapped
    requests.post("http://localhost:3000/api/instance/init")
    print("Instance initialized")
else:
    config: dict = data["config"]
    upgrade: dict = data["upgrade"]
    print(f"Branch: {config['branchName']}")
    print(f"Upgrade available: {upgrade['upgradeAvailable']}")
    print(f"Commits behind: {upgrade['commitsBehind']}")

# 3. Force-check for new upstream commits
check_res = requests.post("http://localhost:3000/api/instance/upgrade/check")
check: dict = check_res.json()

if check["ok"] and check.get("state", {}).get("upgradeAvailable"):
    # 4. Spawn the upgrade task
    upgrade_res = requests.post("http://localhost:3000/api/instance/upgrade")
    task_id: str = upgrade_res.json()["taskId"]
    print(f"Upgrade task spawned: {task_id}")
    print("Navigate to the task view to watch streaming progress")`} />

### Initialize Instance

<EndpointCard method="POST" path="/api/instance/init" description="Idempotent manual re-run of the instance bootstrap. Returns the refreshed instance state after re-running bootstrap.">

<SchemaTable title="Response Body" fields={[
  { name: 'ensureResult', type: 'object', required: true, description: 'Bootstrap result from ensureInstance()' },
  { name: 'config', type: 'object | null', required: true, description: 'Instance configuration after bootstrap' },
  { name: 'guardrails', type: 'object | null', required: true, description: 'Guardrail settings after bootstrap' },
  { name: 'upgrade', type: 'object | null', required: true, description: 'Upgrade state after bootstrap' },
]} />

Re-run bootstrap to recover from a failed initialization or re-apply changed consent settings:

<CodeExample ts={`// Re-run instance bootstrap — safe to call multiple times
const res: Response = await fetch('/api/instance/init', { method: 'POST' });

if (res.ok) {
  const { ensureResult, config }: {
    ensureResult: { created: boolean };
    config: { branchName: string } | null;
  } = await res.json();

  if (ensureResult.created) {
    console.log(\`Instance bootstrapped on branch: \${config?.branchName}\`);
  } else {
    console.log('Instance already initialized — guardrails re-applied');
  }
} else {
  const { error }: { error: string } = await res.json();
  console.error(\`Bootstrap failed: \${error}\`);
}`} python={`import requests

# Re-run instance bootstrap — safe to call multiple times
response = requests.post("http://localhost:3000/api/instance/init")

if response.ok:
    data: dict = response.json()
    if data["ensureResult"].get("created"):
        branch = (data.get("config") or {}).get("branchName", "unknown")
        print(f"Instance bootstrapped on branch: {branch}")
    else:
        print("Instance already initialized — guardrails re-applied")
else:
    print(f"Bootstrap failed: {response.json()['error']}")`} />

**Errors:** `500` — Bootstrap failure (check logs for permission or git errors)

</EndpointCard>

### Spawn Upgrade Task

<EndpointCard method="POST" path="/api/instance/upgrade" description="Spawn an upgrade-assistant task that performs a guided git merge of upstream commits. Returns 202 Accepted with the task ID immediately — the merge runs asynchronously. Requires upgradeAvailable to be true. Rejects with 409 if the instance is not initialized or no upgrade is available.">

<SchemaTable title="Response Body" fields={[
  { name: 'taskId', type: 'string (UUID)', required: true, description: 'ID of the spawned upgrade task — navigate to this task to watch progress' },
]} />

The upgrade task uses the `upgrade-assistant` agent profile. The task description contains the branch name, number of commits behind, and data directory as context variables that the profile's SKILL.md references. The task runs the standard merge flow: fetch upstream, merge, handle conflicts interactively, and rollback on any failure.

<CodeExample ts={`// Spawn the upgrade task — navigate to the task view for streaming progress
const res: Response = await fetch('/api/instance/upgrade', { method: 'POST' });

if (res.status === 202) {
  const { taskId }: { taskId: string } = await res.json();
  console.log(\`Upgrade task: \${taskId}\`);
  // Navigate to /tasks/\${taskId} to watch the merge progress
} else if (res.status === 409) {
  const { error }: { error: string } = await res.json();
  console.log(\`Cannot upgrade: \${error}\`);
}`} python={`import requests

# Spawn the upgrade task — navigate to the task view for streaming progress
response = requests.post("http://localhost:3000/api/instance/upgrade")

if response.status_code == 202:
    task_id: str = response.json()["taskId"]
    print(f"Upgrade task: {task_id}")
    # Navigate to /tasks/{task_id} to watch the merge progress
elif response.status_code == 409:
    print(f"Cannot upgrade: {response.json()['error']}")`} />

Example response:

```json
{ "taskId": "a1b2c3d4-e5f6-7890-abcd-ef1234567890" }
```

**Errors:**
- `409` — Instance not initialized (call `POST /api/instance/init` first) or `upgradeAvailable` is false
- `500` — Internal failure persisting the task or upgrade state

</EndpointCard>

<EndpointCard method="POST" path="/api/instance/upgrade/check" description="Force-run the upgrade availability poller. Rate-limited to one run per ~5 minutes via a shared lock file. <SchemaTable title="Response Body (200 — poller ran)" fields={[

<SchemaTable title="Response Body (202 — poller skipped)" fields={[

  { name: 'error', type: 'string', required: false, description: 'Error message if the poller threw' },

Trigger an immediate upgrade check — useful after the user clicks "Check now" in the settings UI:

<CodeExample ts={`// Force an upgrade check and display the result

}`} python={`import requests

Poll upgrade status from a client component — e.g. to show a badge or refresh the pre-flight modal:

<CodeExample ts={`// Poll upgrade status from a client component
interface UpgradeStatus {
  devMode: boolean;
  upgradeAvailable: boolean;
  commitsBehind: number;
  lastPolledAt: number | null;
  lastUpgradeTaskId: string | null;
  pollFailureCount: number;
}

const status: UpgradeStatus = await fetch('/api/instance/upgrade/status')
  .then((r: Response) => r.json());

  console.log('Dev mode — upgrade status suppressed');

  console.log(\`Upgrade available: \${status.commitsBehind} commits behind\`);
  if (status.lastUpgradeTaskId) {
    console.log(\`Previous upgrade task: \${status.lastUpgradeTaskId}\`);
  }
} else {
  const lastPoll = status.lastPolledAt
    ? new Date(status.lastPolledAt * 1000).toLocaleString()
    : 'never';
  console.log(\`Up to date. Last checked: \${lastPoll}\`);
}`} python={`import requests
from datetime import datetime

# Poll upgrade status from a client component
response = requests.get("http://localhost:3000/api/instance/upgrade/status")
status: dict = response.json()

if status["devMode"]:
    print("Dev mode — upgrade status suppressed")
elif status["upgradeAvailable"]:
    print(f"Upgrade available: {status['commitsBehind']} commits behind")

        print(f"Previous upgrade task: {status['lastUpgradeTaskId']}")
else:
    if status["lastPolledAt"]:
        last_poll = datetime.fromtimestamp(status["lastPolledAt"]).strftime("%Y-%m-%d %H:%M")
    else:
        last_poll = "never"

## Upgrade State Schema

| Field | Type | Description |
|-------|------|-------------|
| upgradeAvailable | boolean | True when `commitsBehind > 0` and last poll succeeded |
| commitsBehind | number | Commits local branch is behind `origin/main` |
| lastPolledAt | number \| null | Unix timestamp of last successful `git rev-list` check |
| lastSuccessfulUpgradeAt | number \| null | Unix timestamp of last completed upgrade |
| lastUpgradeTaskId | string \| null | UUID of the most recently spawned upgrade task |
| pollFailureCount | number | Consecutive poll failures — resets to 0 on success |
| lastPollError | string \| null | Error message from the most recent failed poll |

## Dev Mode

When `AINATIVE_DEV_MODE=true` is set in the environment or the `.git/ainative-dev-mode` sentinel file is present, the Instance API operates in dev mode:

- `GET /api/instance/config` — returns `{ devMode: true, config: null, guardrails: null, upgrade: null }`
- `GET /api/instance/upgrade/status` — returns a synthetic state with `upgradeAvailable: false`
- `POST /api/instance/upgrade` and `POST /api/instance/upgrade/check` — are not blocked but return early without writing state

This prevents stale instance rows written during local testing from surfacing in production UIs when the dev repo is opened.
